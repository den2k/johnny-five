# LED - RGB (Common Anode)

Run with:
```bash
node eg/led-rgb-anode.js
```
<!--remove-end-->

```javascript
var five = require("johnny-five");
var board = new five.Board();

board.on("ready", function() {
  var anode = new five.Led.RGB({
    pins: {
      red: 3,
      green: 5,
      blue: 6
    },
    isAnode: true
  });

  anode.blink();

  this.repl.inject({
    anode: anode
  });

});

```
